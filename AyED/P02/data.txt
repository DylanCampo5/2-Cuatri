    template<class T>
T matrix_t<T>::sum_diagonal_sec() const
{
  assert(get_m() == get_n()); // Comprueba que la matriz es cuadrada
  T sum = 0; // Crea con lo que vamos a sumar los elementos
  for (int i = 1, j = get_n(); i <= get_m(); i++,j--) {
    sum += at(i, j);
  }
  return sum;
}
